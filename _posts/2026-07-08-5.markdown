### 1번 시도:

import csv

def load_tasks(filename):
    f = open(filename, 'r')
    lines = f.readlines()
    f.close()

    result = []

    lines.pop(0)

    for l in lines:
        l = l.strip()
        result.append(l.split(","))

    return result
