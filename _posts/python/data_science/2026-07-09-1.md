## 1번 시도:

{% highlight python %}
import kagglehub
import pandas as pd
from pandas import DataFrame as df
import numpy as np

path = kagglehub.dataset_download("jessemostipak/hotel-booking-demand") + "/hotel_bookings.csv"
df = pd.read_csv(path)
df.head()
df.info()

# 취소
canceled_list = df.loc[df['reservation_status'].isin(['Canceled', 'No-Show'])]
print(f"{round((len(canceled_list) / len(df)) * 100)}%")

# 신규 고객이면서 곧바로 취소
canceled_immediately_list = canceled_list.loc[(canceled_list['is_repeated_guest'] == 0) & (canceled_list['booking_changes'] == 0)]
print(f"{round((len(canceled_immediately_list) / len(canceled_list)) * 100)}%")

# 원인 분석
canceled_list = canceled_list.sort_values(by='lead_time', ascending=False)
print(len(canceled_list.loc[canceled_list['lead_time'] < 300]))
print(canceled_list['lead_time'].describe())
{% endhighlight %}

**Feedback:**
- `len()` 대신 `describe()` 활용하기
- `canceled_immediately_list` 생성 후 미활용
- 무의미한 데이터 출력 및 정렬


## 2번 시도:

{% highlight python %}
import kagglehub
import pandas as pd
from pandas import DataFrame as df
import matplotlib.pyplot as plt
import seaborn as sns

path = kagglehub.dataset_download("jessemostipak/hotel-booking-demand") + "/hotel_bookings.csv"
df = pd.read_csv(path)
df.head()
df.info()

# 취소
canceled_list = df.loc[df['reservation_status'].isin(['Canceled', 'No-Show'])]
print(canceled_list.describe(include='all').T)

# 신규 고객이면서 곧바로 취소
canceled_immediately_list = canceled_list.loc[(canceled_list['is_repeated_guest'] == 0) & (canceled_list['booking_changes'] == 0)]

# 원인 분석
sns.histplot(canceled_immediately_list['lead_time'])
plt.show()

canceled_immediately_list.describe(include='all').T[['count', 'top', 'freq']].dropna(subset=['top', 'freq']).sort_values(by='freq', ascending=False)
{% endhighlight %}
